To host the project
sudo yum install git -y
git clone https://github.com/Dynatrace/easyTravel-Docker.git
cd easyTravel-Docker


download docker==>sudo curl -L "https://github.com/docker/compose/releases/download/1.29.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
install docker==>sudo chmod +x /usr/local/bin/docker-compose
compose docker files=>docker-compose up -d
